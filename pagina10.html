Bubble Sort (Sortare prin bule)

#include 
using namespace std;
int main() {
	int n, i, aux, a[1001], ordonat = 0;
	cin >> n;
	for(i = 1; i <= n; i++)
		cin >> a[i];
	while(ordonat == 0) {
		ordonat = 1;
	for(i = 1; i < n; i++)
		if(a[i] > a[i+1]) {
			aux = a[i];
			a[i] = a[i+1];
			a[i+1] = aux;
			ordonat = 0;
		}
	}
	for(i = 1; i <= n; i++)
		cout << a[i] << " ";
	return 0;
}
