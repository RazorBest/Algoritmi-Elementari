Sortare prin insertie

#include 
using namespace std;
int main() {
	int n, i, a[1001], x, j;
	cin >> n;
	for(i = 1; i <= n; i++)
		cin >> a[i];
	for(i = 2; i <= n; i++){
		x = a[i];
		j = i - 1;
		while(x < a[j] && j > 0) {
			a[j + 1] = a[j];
			j--;
		}
		a[j + 1] = x;
	}
	for(i=1; i<=n; i++)
		cout << a[i] << " ";
	return 0;
}
