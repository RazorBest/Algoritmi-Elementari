Palindromitatea unui numar

#include    
using namespace std;  
int main() {    
	int n, copie, ogl = 0, c;    
	cin >> n;    
	copie = n;    
	while(n != 0){    
		c = n % 10;
		ogl = ogl * 10 + c;
		n = n / 10;
	}
	if(copie == ogl)
		cout<< "Este palindrom.";
	else
		cout<< "Nu este palindrom.";
	return 0;
}
