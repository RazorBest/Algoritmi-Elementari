Primalitatea unui numar

#include     
using namespace std;    
int main() {    
	int n, prim = 1;    
	cin >> n;    
	if(n < 2) prim = 0;    
	if(n != 2 && n % 2 == 0) prim = 0;    
	for(int d = 3; d * d <= n; d+=2)
		if(n % d == 0) prim = 0;
	if(prim)
		cout<< "Este numar prim.";
	else
		cout<< "Nu este numar prim.";
	return 0;
}
