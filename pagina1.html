Cel mai mare divizor comun (CMMDC)

Prin metoda lui Euclid:

#include      
using namespace std;
int main() {     
	int a, b, r;     
	cin >> a >> b;    
	while(b != 0){     
		r = a % b; 
		a = b; 
		b = r;
	} 
	cout << a << " ";
	return 0;
}

Prin scadere:

#include 
using namespace std;
int main() {
	int a, b;
	cin >> a >> b;
	if(a == 0) cout << b;
	if(b == 0) cout << a;
	else { 
		while(a != b)
			if(a > b) a = a - b;
			else b = b - a;
	}
	cout << a << " ";
	return 0;
}
